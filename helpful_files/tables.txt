CREATE TABLE PublicUser
(
    id          INT AUTO_INCREMENT,
    username    VARCHAR(20) NOT NULL,
    location    VARCHAR(20) NOT NULL,
    email       VARCHAR(40) NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE Users
(
    id          INT,
    password    VARCHAR(40),
    PRIMARY KEY (id, password),
    FOREIGN KEY (id) REFERENCES PublicUser(id)
);
